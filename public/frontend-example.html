// React Component Example
import { useState } from 'react';

function PaymentComponent() {
  const [payment, setPayment] = useState(null);

  const createPayment = async (amount, currency) => {
    const response = await fetch('/api/payments', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        amount,
        currency,
        description: 'Payment',
        merchantWallet: 'YOUR_WALLET',
        userId: 'USER_ID'
      })
    });
    
    const data = await response.json();
    setPayment(data);
    return data;
  };

  const payWithWallet = () => {
    if (payment?.paymentUrl) {
      window.location.href = payment.paymentUrl;
    }
  };

  return (
    <div>
      <button onClick={() => createPayment(0.1, 'SOL')}>
        Pay 0.1 SOL
      </button>
      {payment && (
        <div>
          <img src={payment.qrCode} alt="QR Code" />
          <button onClick={payWithWallet}>Pay with Wallet</button>
          <a href={payment.shareUrl}>Share Payment Link</a>
        </div>
      )}
    </div>
  );
}
